git log --oneline : visualizar todos os commits, um em cada linha
git log -p : ver mais como as alterações do commit
git log --author="user_name" : visualizar somente o autor
git log --since=1.month.ago --until=1.day.ago : pesquisar por data
git log --pretty="format:%h %s" : formatar a visualização
